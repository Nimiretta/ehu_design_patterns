// Валидные четырехугольники
Quad sq1 0 0 2 0 2 2 0 2           // квадрат
Quad rect1 0 0 3 0 3 1 0 1         // прямоугольник
Quad itrap1 0 0 4 0 3 2 1 2        // равнобедренная трапеция
Quad trap1 0 0 5 0 4 2 1 2         // произвольная трапеция
Quad quad1 0 0 2 1 1 3 -1 2        // выпуклый четырехугольник
Quad conc1 0 0 3 0 1 1 4 2         // невыпуклый четырехугольник

// Невалидные четырехугольники
Quad bad1 0 0 0 0 1 1 0 1          // совпадающие точки
Quad bad2 0 0 1 1 2 2 3 3          // все точки на одной прямой
Quad bad3 0 0 1 0 2 0 3 4          // три точки на одной прямой (ось X)
Quad bad4 0 0 1 0 1 0 0 1          // три точки совпадают

// Валидные кубы
Cube cube1 0 0 0 2                  // основание на всех осях
Cube cube2 1 1 1 1                  // не пересекает ни одну ось
Cube cube3 0.5 2 2 2                // плоскость YZ (x=0) проходит внутри
Cube cube4 2 0.5 2 2                // плоскость XZ (y=0) проходит внутри
Cube cube5 2 2 0.5 2                // плоскость XY (z=0) проходит внутри

// Невалидные кубы
Cube badcube1 0 0 0 -2             // отрицательная длина ребра
Cube badcube2 0 0 0 0              // нулевая длина ребра
Cube badcube3 0 0 0 aa2            // нечисловая длина ребра
Cube badcube4 0 0 0                // не хватает параметров
Cube badcube5 0 0                  // не хватает параметров
